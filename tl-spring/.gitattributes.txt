# 모든 텍스트 파일은 UTF-8로 저장
* text=auto eol=lf working-tree-encoding=UTF-8

# Java 소스코드 UTF-8 강제
*.java text working-tree-encoding=UTF-8

# XML, properties, yml 등 설정 파일도 UTF-8
*.xml text working-tree-encoding=UTF-8
*.yml text working-tree-encoding=UTF-8
*.properties text working-tree-encoding=UTF-8

# Markdown / txt / sql도 UTF-8
*.md text working-tree-encoding=UTF-8
*.txt text working-tree-encoding=UTF-8
*.sql text working-tree-encoding=UTF-8
